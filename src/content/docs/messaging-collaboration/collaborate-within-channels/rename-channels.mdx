import mobileSelectChannel from '../../../../images/mobile-select-a-channel.jpg';

import mobileViewInfo from '../../../../images/mobile-select-view-info-for-a-channel.jpg';
import mobileEditChannel from '../../../../images/mobile-edit-channel.jpg';
import mobileRenameSave from '../../../../images/mobile-rename-channel-name-and-save.jpg';

يمكن لأي شخص إعادة تسمية القنوات التي ينتمي إليها، ما لم يقم مسؤول النظام بـ [تقييد الأذونات](#) للقيام بذلك باستخدام الأذونات المتقدمة.

<Tabs>
  <TabItem label="الويب/سطح المكتب" icon="laptop">
    <Steps>
    1. اختر اسم القناة في أعلى اللوحة المركزية للوصول إلى القائمة المنسدلة.
    2. اختر **إعدادات القناة (Channel Settings)**.
    3. قم بتحديث المعلومات التالية في النافذة المنبثقة:
    </Steps>

    - **اسم القناة (Channel name):** الاسم الذي يظهر في واجهة مستخدم Mattermost لجميع الأعضاء. أدخل اسماً جديداً يعبر عن محتوى القناة.
    - **عنوان URL للقناة (Channel URL):** الرابط المباشر للوصول إلى القناة عبر المتصفح. اختر **تحرير (Edit)** لتعديل الرابط، ثم اضغط **تم (Done)**.

    <Aside type="note" title="تنبيه بخصوص الروابط">
      إذا قام مسؤول النظام بتمكين عناوين URL المجهولة (بداية من النسخة <Badge text="v11.6.0" variant="note" />)، فسيتم تعيين الروابط تلقائياً ولن تعكس اسم القناة بالضرورة.
    </Aside>

    

    ---
    **مثال:**
    يمكن تسمية القناة `تصميم تجربة المستخدم (UX Design)` ويكون الرابط الخاص بها:
    `https://community.mattermost.com/core/channels/ux-design`
  </TabItem>

  <TabItem label="الهاتف المحمول" icon="mobile">
    <Steps>
    1. اضغط على اسم القناة التي ترغب في إعادة تسميتها من القائمة.
       <Image src={mobileSelectChannel} alt="Step 1" style={{ height: '450px', width: 'auto', borderRadius: '15px', border: '1px solid #ddd', display: 'block', margin: '2rem auto' }} />

    2. اضغط على أيقونة **المزيد (More)** (ثلاث نقاط رأسية) في الزاوية العلوية اليمنى.
       <Image src={mobileMoreOptions} alt="Step 2" style={{ height: '450px', width: 'auto', borderRadius: '15px', border: '1px solid #ddd', display: 'block', margin: '2rem auto' }} />

    3. من القائمة الظاهرة، اختر **عرض المعلومات (View info)**.
       <Image src={mobileViewInfo} alt="Step 3" style={{ height: '450px', width: 'auto', borderRadius: '15px', border: '1px solid #ddd', display: 'block', margin: '2rem auto' }} />

    4. اضغط على خيار **تحرير القناة (Edit Channel)** لبدء التعديل.
       <Image src={mobileEditChannel} alt="Step 4" style={{ height: '450px', width: 'auto', borderRadius: '15px', border: '1px solid #ddd', display: 'block', margin: '2rem auto' }} />

    5. قم بتعديل البيانات المطلوبة (الاسم، الغرض، أو الترويسة) ثم اضغط على **حفظ (Save)**.
    </Steps>

    <Image src={mobileRenameSave} alt="Save Changes" style={{ height: '450px', width: 'auto', borderRadius: '15px', border: '1px solid #ddd', display: 'block', margin: '2rem auto' }} />

    <Aside type="tip">
      تذكر أن تعديل **اسم القناة (Name)** سيؤدي لتغيير ظهورها لدى جميع أعضاء الفريق فوراً.
    </Aside>
  </TabItem>
</Tabs>
